# Destroy

`useDestroy` allows to perform `DELETE` (destroy) requests.

## Definition

```ts filename="Import statement (TypeScript)"
import { useDestroy } from "tqa/hooks/crud";
```

```ts filename="Signature (TypeScript)"
const useDestroy: <TResponse = unknown, TParams = unknown, TError = unknown>(
  url: Endpoint,
  config: UseDestroyOptions<TResponse, TParams, TError>,
  queryClient?: QueryClient
) => UseDestroyResult<TResponse, TError>;
```

## Examples

### Destroy (`DELETE`)

```tsx filename="TypeScript (TSX)"
import { useDestroy } from "tqa/hooks/crud";

export interface DeleteBookButtonProps {
  id: number;
}

export const DeleteBookButton = ({ id }: DeleteBookButtonProps) => {
  const destroyer = useDestroy(`/api/book/${id}`, {
    reactQuery: { mutationKey: ["DESTROY_BOOK", id] },
  });

  return <button onClick={() => void destroyer.mutateAsync()}>Destroy</button>;
};
```
